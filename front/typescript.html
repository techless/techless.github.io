# 函数

// 正常函数
const fn = function (name: string, age: number) {
  return name + age
}
const a = fn('张三'， 18)

// 默认参数
const fn = function (name: string, age: number = 18) {
  return name + age
}
const a = fn('张三')

// 可不传age, 为undefined
const fn = function (name: string, age?: number) {
  return name + age
}
const a = fn('张三')

// 接口
interface User{
  name: string,
  age: number
}

const fn = function (user: User): User {
  return user
}

let a = fn({
  name: '张三',
  age: 18
})

// 重载
function fn(params: number): void
function fn(params: string, params2:number): void
function fn(params: any, params2?: any):any

# 类型

// 联合类型
let phone: number | string = 18888888888
let phone: number | string = '18888888888'

let fn = function (type: number | boolean): boolean {
  return !!type
}

// 交叉类型
interface People {
  name: string
  age: number
}

interface Other {
  sex: number
}

const people = (man: People & Other): void => {
  console.log('man', man)
}

people({
  name: 'John',
  age: 13,
  sex: 1,
})

// 类型断言
let fn = function (num: Number | String): void {
  console.log((num as string).length)
}

interface A {
  run: string
}

interface B {
  build: string
}

let fn = (type: A | B): void => {
  console.log((<A>type).run)
  console.log((type as A).run)
}

# 内置对象

  • TypeScript 中描述类型要用 小写，比如 boolean、number、string 等；
  • 大写开头的如 Boolean、Number、String 代表的是 JavaScript 的构造函数:
// 正则
const regexp: RegExp = /\d/
// 日期
const date: Date = new Date()
const date: number = new Date().getTime()
// 错误
const error: Error = new Error('error')
const bool: Boolean = new Boolean(true)
const string: String = new String('123')

// DOM
const list: NodeList = document.querySelectorAll('#list li')
