export ZSH="$HOME/.oh-my-zsh"

# oh-my-zsh

    git
    zsh-autosuggestions
    zsh-syntax-highlighting
    zsh-z

# nvm

# proxy on
function proxy_on() {
    export no_proxy="localhost,127.0.0.1,localaddress,.localdomain.com"
    export http_proxy="http://127.0.0.1:19180"
    export https_proxy=$http_proxy
    export all_proxy=socks5://127.0.0.1:19181 # or this line
    curl www.google.com
    echo -e "\n"
    echo -e "\033[32m已开启代理\033[0m"
}

# proxy off
function proxy_off(){
    unset no_proxy
    unset http_proxy
    unset https_proxy
    unset all_proxy
    echo -e "已关闭代理"
}
